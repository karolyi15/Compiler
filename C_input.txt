//Virtual Reality-Rutina de prueba
import _Main;
declare steps;

procedure virtualReality(steps)
    begin

        steps=call inc(steps,5);
        call move(steps);
        call delay(5);

        steps=call dec(steps,1);
        call object(steps);
        call delay(5);

        call temperature(2);
        call delay(5);

    end;

procedure main()
    begin
        call virtualReality(steps);
    end;
